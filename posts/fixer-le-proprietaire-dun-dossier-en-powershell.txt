	$folders = Get-ChildItem | where {$_.mode -match "d"} | select Psdrive, parent, name, fullname
	foreach ($folder in $folders) { 
		$subfolder= Get-ChildItem -Path $folder.name | where {$_.mode -match "d"} | select PSdrive, parent, name, fullname
		write-host $folder.name  - $subfolder.name - (get-acl $subfolder.fullname).owner
	}

		}
		catch
		{
		write-host $folder.name  - $subfolder.name  - $_.exception.message
